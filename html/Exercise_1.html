steps = logspace(-2, 2, 1000) * 1i;

[mag, phase] = mybode(1, [2 1], 0, steps);

figure(1);
subplot(2, 1, 1);
semilogx(abs(steps), mag);
subplot(2, 1, 2);
semilogx(abs(steps), phase);


function [mag, phase] = mybode(a, b, Tt, w)
    if isempty(Tt)
        Tt = 0;
    end

    if isempty(w)
        w = logspace(-2, 2, 1000) * sqrt(-1);
    end

    g = polyval(a, w) ./ polyval(b, w) .* exp(-Tt * w);

    mag = 20 * log10(abs(g));
    phase = angle(g);
end
